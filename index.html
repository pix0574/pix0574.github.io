###获取猫眼电影top100
1、并用正则提取数据。
2、实现翻页功能。

import json
import requests
from requests.exceptions import RequestException
import re
import time


def get_one_page(url):
    try:
        headers = {
            'User-Agent': ',
        }
        response = requests.get(url, headers=headers)
        if response.status_code == 200:
            return response.text
        return None
    except RequestException:
        return None


def parse_one_page(html):
    pattern = re.compile('<dd>.*?board-index.*?>(\d+)</i>.*?data-src="(.*?)".*?name"><a'
                         + '.*?>(.*?)</a>.*?star">(.*?)</p>.*?releasetime">(.*?)</p>'
                         + '.*?integer">(.*?)</i>.*?fraction">(.*?)</i>.*?</dd>', re.S)
    items = re.findall(pattern, html)
    for item in items:
        yield {
            'index': item[0],
            'image': item[1],
            'title': item[2],
            'actor': item[3].strip()[3:],
            'time': item[4].strip()[5:],
            'score': item[5] + item[6]
        }


def write_to_file(content):
    with open('result.txt', 'a', encoding='utf-8') as f:
        f.write(json.dumps(content, ensure_ascii=False) + '\n')


def main(offset):
    url = 'http://maoyan.com/board/4?offset=' + str(offset)
    html = get_one_page(url)
    for item in parse_one_page(html):
        print(item)
        write_to_file(item)


if __name__ == '__main__':
    for i in range(10):
        main(offset=i * 10)
        time.sleep(1)

####获取并打印cookie的值

import requests
r= requests.get('url')
print(r.cookies)

for key,value in r.cookies.items():
print(key + '=' + value)

#条件表达式，比较返回码和内置的返回码，保证请求得到了正常响应。
import requests
r = requests.get('url')

exit() if not r.status_code == requests.codes.forbidden else print('Request Successfully')

####添加header头和参数
import requests

headers = {
    'ua':'',
    'cookie':''}

data = {
    'name' : 'kevin',
    'age' : '22'
    }

#设置代理
proxies = {}
#代理 HTTP Basic Auth认证
proxies = {"http":"http://user:password@ip"}    

#超时设置
r = requests.get('url',timeout=1)

#身份认证
from requests.auth import HTTPBasicAuth
r = requests.get('url',auth=('username','password'))

r=requests.get('url',headers=headers,params = data,proxies = proxies，timeout=1，auth=('username','password'))

####爬去二进制图片
import requests
r = requests.get(‘file’)
with open(‘favion.ico’,’wb’)as f:
f.write(r.content)

####文件上传
import requests
files = {‘file’:open(‘favicon.ico’,’rb’)}
r = requests.post(‘url’,files = files)
print(r.text)

####会话维持

#session对象，可以维持一个网站的多个页面
import requests
s = requests.Session()
s.get('url')
r = s.get('url')
print(r.text)

####证书加密
import requests
from requests.packages import urllib3

#取消报警
urllib3.disable_warnings()

#取消认证
r = requests.get('https://www.12306.cn',verify=False)
print(r.status_code)
